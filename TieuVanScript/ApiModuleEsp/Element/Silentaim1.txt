return function(AimPage, SilentAimAPI)
	local RWSection = AimPage:Section({
		Name = "Round War Silent Aim",
		Description = "advanced silent aim",
		Icon = "10709818534",
		Side = 1
	})

	RWSection:Toggle({
		Name = "Enable",
		Flag = "RWSilentAimEnable",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({SilentAim = Value})
			SilentAimAPI:Toggle(Value)
		end
	})

	RWSection:Slider({
		Name = "FOV Size",
		Flag = "RWSilentAimFOV",
		Min = 1,
		Max = 500,
		Default = 150,
		Decimals = 1,
		Suffix = "px",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({FOV = Value})
		end
	})

	RWSection:Label("FOV Color"):Colorpicker({
		Name = "FOV Color",
		Flag = "RWSilentAimFOVColor",
		Default = Color3.fromRGB(255, 255, 255),
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({FOVColor = Value})
		end
	})

	RWSection:Slider({
		Name = "Hit Chance",
		Flag = "RWSilentAimHitChance",
		Min = 0,
		Max = 100,
		Default = 100,
		Decimals = 1,
		Suffix = "%",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({HitChance = math.floor(Value)})
		end
	})

	RWSection:Dropdown({
		Name = "Target Mode",
		Flag = "RWSilentAimTargetMode",
		Default = "Players",
		Items = {"NPCs", "Players", "Both"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAPI:UpdateConfig({TargetMode = Value[1]})
			else
				SilentAimAPI:UpdateConfig({TargetMode = Value})
			end
		end
	})

	RWSection:Dropdown({
		Name = "Aim Part",
		Flag = "RWSilentAimPart",
		Default = "Head",
		Items = {"Head", "Torso"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAPI:UpdateConfig({AimPart = Value[1]})
			else
				SilentAimAPI:UpdateConfig({AimPart = Value})
			end
		end
	})

	RWSection:Dropdown({
		Name = "Target Priority",
		Flag = "RWSilentAimTargetPriority",
		Default = "Crosshair",
		Items = {"Crosshair", "Distance", "LowestHP"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAPI:UpdateConfig({TargetPriority = Value[1]})
			else
				SilentAimAPI:UpdateConfig({TargetPriority = Value})
			end
		end
	})

	RWSection:Toggle({
		Name = "Team Check",
		Flag = "RWSilentAimTeamCheck",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({TeamCheck = Value})
		end
	})

	RWSection:Toggle({
		Name = "Team Check (NPCs)",
		Flag = "RWSilentAimTeamCheckNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({TeamCheckForNPCs = Value})
		end
	})

	RWSection:Toggle({
		Name = "Wall Check",
		Flag = "RWSilentAimWallCheck",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({VisibleCheck = Value})
		end
	})

	RWSection:Toggle({
		Name = "Lock Target",
		Flag = "RWSilentAimLockTarget",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({StickyAim = Value})
		end
	})

	RWSection:Toggle({
		Name = "Prediction",
		Flag = "RWSilentAimPrediction",
		Default = true,
		Callback = function(Value)
			if Value then
				SilentAimAPI:UpdateConfig({Prediction = 0.165})
			else
				SilentAimAPI:UpdateConfig({Prediction = 0})
			end
		end
	})

	RWSection:Slider({
		Name = "Prediction Factor",
		Flag = "RWSilentAimPredictionFactor",
		Min = 0.01,
		Max = 1,
		Default = 0.165,
		Decimals = 0.01,
		Suffix = "x",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({Prediction = Value})
		end
	})

	RWSection:Slider({
		Name = "Update Rate",
		Flag = "RWSilentAimUpdateRate",
		Min = 0.01,
		Max = 1,
		Default = 0.1,
		Decimals = 0.01,
		Suffix = "s",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({UpdateRate = Value})
		end
	})

	RWSection:Toggle({
		Name = "Aggressive NPC Detection",
		Flag = "RWSilentAimAggressiveNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({AggressiveNPCDetection = Value})
		end
	})

	RWSection:Toggle({
		Name = "Debug NPCs",
		Flag = "RWSilentAimDebugNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({DebugNPCs = Value})
		end
	})
end

return function(AimPage, SilentAimAPI)
	local SilentAimSection = AimPage:Section({
		Name = "Silent Aim",
		Description = "silent aim",
		Icon = "10709818534",
		Side = 1
	})

	SilentAimSection:Toggle({
		Name = "Enable",
		Flag = "SilentAimEnable",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({Enabled = Value})
			SilentAimAPI:Toggle(Value)
		end
	})

	SilentAimSection:Slider({
		Name = "FOV Size",
		Flag = "SilentAimFOV",
		Min = 1,
		Max = 500,
		Default = 150,
		Decimals = 1,
		Suffix = "px",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({FOVSize = Value})
		end
	})

	SilentAimSection:Label("FOV Color"):Colorpicker({
		Name = "FOV Color",
		Flag = "SilentAimFOVColor",
		Default = Color3.fromRGB(0, 255, 0),
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({FOVColor = Value})
		end
	})

	SilentAimSection:Slider({
		Name = "Max Range",
		Flag = "SilentAimRange",
		Min = 1,
		Max = 10000,
		Default = 5000,
		Decimals = 1,
		Suffix = "m",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({MaxRange = Value})
		end
	})

	SilentAimSection:Dropdown({
		Name = "Target Mode",
		Flag = "SilentAimTargetMode",
		Default = "NPCs",
		Items = {"NPCs", "Players", "Both"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAPI:UpdateConfig({TargetMode = Value[1]})
			else
				SilentAimAPI:UpdateConfig({TargetMode = Value})
			end
		end
	})

	SilentAimSection:Dropdown({
		Name = "Aim Part",
		Flag = "SilentAimPart",
		Default = "Head",
		Items = {"Head", "UpperTorso", "LowerTorso"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAPI:UpdateConfig({AimPart = Value[1]})
			else
				SilentAimAPI:UpdateConfig({AimPart = Value})
			end
		end
	})

	SilentAimSection:Toggle({
		Name = "Team Check",
		Flag = "SilentAimTeamCheck",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({TeamCheck = Value})
		end
	})

	SilentAimSection:Toggle({
		Name = "Wall Check",
		Flag = "SilentAimWallCheck",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({WallCheck = Value})
		end
	})

	SilentAimSection:Toggle({
		Name = "Lock Target",
		Flag = "SilentAimLockTarget",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({LockTarget = Value})
		end
	})

	SilentAimSection:Toggle({
		Name = "Prediction",
		Flag = "SilentAimPrediction",
		Default = true,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({PredictionEnabled = Value})
		end
	})

	SilentAimSection:Slider({
		Name = "Prediction Factor",
		Flag = "SilentAimPredictionFactor",
		Min = 0.01,
		Max = 1,
		Default = 0.165,
		Decimals = 0.01,
		Suffix = "x",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({PredictionFactor = Value})
		end
	})

	SilentAimSection:Slider({
		Name = "Hit Chance",
		Flag = "SilentAimHitChance",
		Min = 0,
		Max = 100,
		Default = 100,
		Decimals = 1,
		Suffix = "%",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({HitChance = math.floor(Value)})
		end
	})

	SilentAimSection:Slider({
		Name = "Vertical Offset",
		Flag = "SilentAimVerticalOffset",
		Min = -1,
		Max = 1,
		Default = 0,
		Decimals = 0.1,
		Suffix = "m",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({OffsetY = Value})
		end
	})

	SilentAimSection:Slider({
		Name = "FOV Vertical Offset",
		Flag = "SilentAimFOVVerticalOffset",
		Min = -100,
		Max = 100,
		Default = 0,
		Decimals = 1,
		Suffix = "px",
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({FOVOffsetY = Value})
		end
	})

	SilentAimSection:Toggle({
		Name = "Aggressive NPC Detection",
		Flag = "SilentAimAggressiveNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({AggressiveNPCDetection = Value})
		end
	})

	SilentAimSection:Toggle({
		Name = "Debug NPCs",
		Flag = "SilentAimDebugNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAPI:UpdateConfig({DebugNPCs = Value})
		end
	})
end

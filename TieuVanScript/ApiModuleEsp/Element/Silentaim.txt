return function(AimPage, SilentAimAdvancedAPI)
	local SASection = AimPage:Section({
		Name = "Silent Aim Advanced",
		Description = "advanced silent aim with fov options",
		Icon = "10709818534",
		Side = 1
	})

	SASection:Toggle({
		Name = "Enable",
		Flag = "SAEnable",
		Default = false,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({SilentAim = Value})
			SilentAimAdvancedAPI:Toggle(Value)
		end
	})

	SASection:Slider({
		Name = "FOV Size",
		Flag = "SAFOV",
		Min = 1,
		Max = 300,
		Default = 150,
		Decimals = 1,
		Suffix = "%",
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({FOV = Value})
		end
	})

	SASection:Slider({
		Name = "FOV Scale",
		Flag = "SAFOVScale",
		Min = 0.1,
		Max = 3,
		Default = 1,
		Decimals = 0.1,
		Suffix = "x",
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({FOVScale = Value})
		end
	})

	SASection:Toggle({
		Name = "Screen Percentage Mode",
		Flag = "SAScreenPercentage",
		Default = true,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({UseScreenPercentage = Value})
		end
	})

	SASection:Label("FOV Color"):Colorpicker({
		Name = "FOV Color",
		Flag = "SAFOVColor",
		Default = Color3.fromRGB(255, 255, 255),
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({FOVColor = Value})
		end
	})

	SASection:Slider({
		Name = "Hit Chance",
		Flag = "SAHitChance",
		Min = 0,
		Max = 100,
		Default = 100,
		Decimals = 1,
		Suffix = "%",
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({HitChance = math.floor(Value)})
		end
	})

	SASection:Dropdown({
		Name = "Target Mode",
		Flag = "SATargetMode",
		Default = "Players",
		Items = {"NPCs", "Players", "Both"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAdvancedAPI:UpdateConfig({TargetMode = Value[1]})
			else
				SilentAimAdvancedAPI:UpdateConfig({TargetMode = Value})
			end
		end
	})

	SASection:Dropdown({
		Name = "Aim Part",
		Flag = "SAAimPart",
		Default = "Head",
		Items = {"Head", "Torso"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAdvancedAPI:UpdateConfig({AimPart = Value[1]})
			else
				SilentAimAdvancedAPI:UpdateConfig({AimPart = Value})
			end
		end
	})

	SASection:Dropdown({
		Name = "Target Priority",
		Flag = "SATargetPriority",
		Default = "Crosshair",
		Items = {"Crosshair", "Distance", "LowestHP"},
		Multi = false,
		Callback = function(Value)
			if type(Value) == "table" then
				SilentAimAdvancedAPI:UpdateConfig({TargetPriority = Value[1]})
			else
				SilentAimAdvancedAPI:UpdateConfig({TargetPriority = Value})
			end
		end
	})

	SASection:Toggle({
		Name = "Team Check",
		Flag = "SATeamCheck",
		Default = false,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({TeamCheck = Value})
		end
	})

	SASection:Toggle({
		Name = "Team Check (NPCs)",
		Flag = "SATeamCheckNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({TeamCheckForNPCs = Value})
		end
	})

	SASection:Toggle({
		Name = "Wall Check",
		Flag = "SAWallCheck",
		Default = true,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({VisibleCheck = Value})
		end
	})

	SASection:Toggle({
		Name = "Lock Target",
		Flag = "SALockTarget",
		Default = true,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({StickyAim = Value})
		end
	})

	SASection:Toggle({
		Name = "Prediction",
		Flag = "SAPrediction",
		Default = true,
		Callback = function(Value)
			if Value then
				SilentAimAdvancedAPI:UpdateConfig({Prediction = 0.165})
			else
				SilentAimAdvancedAPI:UpdateConfig({Prediction = 0})
			end
		end
	})

	SASection:Slider({
		Name = "Prediction Factor",
		Flag = "SAPredictionFactor",
		Min = 0.01,
		Max = 1,
		Default = 0.165,
		Decimals = 0.01,
		Suffix = "x",
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({Prediction = Value})
		end
	})

	SASection:Slider({
		Name = "Update Rate",
		Flag = "SAUpdateRate",
		Min = 0.01,
		Max = 1,
		Default = 0.1,
		Decimals = 0.01,
		Suffix = "s",
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({UpdateRate = Value})
		end
	})

	SASection:Toggle({
		Name = "Bullet Teleport",
		Flag = "SABulletTeleport",
		Default = false,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({BulletTeleport = Value})
		end
	})

	SASection:Toggle({
		Name = "Aggressive NPC Detection",
		Flag = "SAAggressiveNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({AggressiveNPCDetection = Value})
		end
	})

	SASection:Toggle({
		Name = "Debug NPCs",
		Flag = "SADebugNPC",
		Default = false,
		Callback = function(Value)
			SilentAimAdvancedAPI:UpdateConfig({DebugNPCs = Value})
		end
	})
end
